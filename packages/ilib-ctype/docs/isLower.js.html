// !data ctype_l

import { JSUtils } from "ilib-common";

import { inRange } from "./CType.js";

import { ctype_l } from "./ctype_l.js";

/**
 * Return whether or not the first character is lower-case. For alphabetic
 * characters in scripts that do not make a distinction between upper- and
 * lower-case, this function always returns true.<p>
 *
 * @static
 * @param {string|IString|number} ch character or code point to examine
 * @return {boolean} true if the first character is lower-case.
 */
export default function isLower(ch) {
    var num;
    switch (typeof(ch)) {
        case 'number':
            num = ch;
            break;
        case 'string':
            num = JSUtils.toCodePoint(ch, 0);
            break;
        case 'undefined':
            return false;
        default:
            num = ch._toCodePoint(0);
            break;
    }

    return inRange(num, 'Ll', ctype_l);
};
